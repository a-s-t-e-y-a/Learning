Hint #1

Grabbing a snapshot of the current data makes it easy to revert back to the original state if there is an error. It also gives gives you data to operate on to perform your optimistic update. Before we can grab a snapshot, we need to cancel any pending queries for the books query. This will ensure that the data isn't updated after we get our snapshot.

function useCheckoutBook(book) {
  const queryClient = useQueryClient();


  return useMutation({
    mutationFn: () => checkoutBook(book.id),
    onMutate: async () => {
      await queryClient.cancelQueries({
        queryKey: ["books"]
      });


      const snapshot = {
        myBooks: queryClient.getQueryData(["books", "my-books"]),
        bookDetail: queryClient.getQueryData(["books", "detail", book.id]),
      };
    },
  });
}
Hint #2

In order to perform this optimistic update, we need to update the cache data in two places. First, we need to add the book to the my-books query. Second, we need to update the detail query for the book to reflect the new state of the book.

function useCheckoutBook(book) {
  const queryClient = useQueryClient();


  return useMutation({
    mutationFn: () => checkoutBook(book.id),
    onMutate: async () => {
      // Add the book to the my-books query
      queryClient.setQueryData(["books", "my-books"], (previousBooks) =>
        previousBooks ? [...previousBooks, book] : undefined
      );


      // Update the detail query for the book
      queryClient.setQueryData(["books", "detail", book.id], (previousBook) =>
        previousBook
          ? {
              ...previousBook,
              isCheckedOutByUser: true,
              availableCopies: previousBook.availableCopies - 1,
            }
          : undefined
      );
    },
  });
}
Hint #3

If there is an error, we need to revert the cache back to the original state. We can do this by returning a function from the onMutate callback that will be called if there is an error. Then, in the onError callback, we can call the rollback function to revert the cache back to the original state. Finally, we can invalidate the books query to refetch the data from the server, to make sure we have the latest data.

function useCheckoutBook(book) {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: () => checkoutBook(book.id),
    onMutate: async () => {
      await queryClient.cancelQueries({
        queryKey: ["books"]
      });
      
      const snapshot = {
        myBooks: queryClient.getQueryData(["books", "my-books"]),
        bookDetail: queryClient.getQueryData(["books", "detail", book.id]),
      };
      // This is the "rollback" function that will be called if there is an error
      return () => {
        queryClient.setQueryData(["books", "my-books"], snapshot.myBooks);
        queryClient.setQueryData(
          ["books", "detail", book.id],
          snapshot.bookDetail
        );
      };
    },
    onError: (_err, _variables, rollback) => {
      console.log(err);
      rollback?.();
    },
    onSettled: () => {
      return queryClient.invalidateQueries({ queryKey: ["books"] });
    },
  });
}

export function useReturnBook(book) {
  const queryClient = useQueryClient();


  return useMutation({
    mutationFn: () => returnBook(book.id),
    onMutate: async () => {
      await queryClient.cancelQueries({
        queryKey: ["books"]
      });


      const snapshot = {
        myBooks: queryClient.getQueryData(["books", "my-books"]),
        bookDetail: queryClient.getQueryData(["books", "detail", book.id])
      };


      queryClient.setQueryData(["books", "my-books"], (previousBooks) =>
        previousBooks
          ? previousBooks.filter((b) => b.id !== book.id)
          : undefined
      );


      queryClient.setQueryData(["books", "detail", book.id], (previousBook) =>
        previousBook
          ? {
              ...previousBook,
              isCheckedOutByUser: false,
              availableCopies: previousBook.availableCopies + 1
            }
          : undefined
      );


      return () => {
        queryClient.setQueryData(["books", "my-books"], snapshot.myBooks);
        queryClient.setQueryData(
          ["books", "detail", book.id],
          snapshot.bookDetail
        );
      };
    },
    onError: (_err, _variables, rollback) => {
      rollback?.();
    },
    onSettled: () => {
      return queryClient.invalidateQueries({ queryKey: ["books"] });
    }
  });
}
