Hint #1

The refetchInterval option allows you to specify how often the query should be refetched. You can use it like this:

function useActivities() {
  return useQuery({
    queryKey: ["activity"],
    queryFn: getActivity,
    refetchInterval: 5000
  });
}
Hint #2

React query returns a dataUpdatedAt property that you can use to get the timestamp of the last data fetch. You can use this to display the "Latest activity as of" message.

const { data, dataUpdatedAt } = useActivities();
formatDate(dataUpdatedAt)

import * as React from "react";
import { useQuery } from "@tanstack/react-query";
import { getActivity, formatDate } from "./utils";
import ActivityList from "./ActivityList";
import Loader from "./Loader";


function useActivities() {
  return useQuery({
    queryKey: ["activity"],
    queryFn: getActivity,
    refetchInterval: 5000
  });
}


function ActivityFeed() {
  const { data, dataUpdatedAt, status } = useActivities();


  if (status === "pending") {
    return <Loader />;
  }


  return (
    <section className="latest-activity">
      <h2>Latest activity as of {formatDate(dataUpdatedAt)}</h2>
      <ActivityList data={data} />
    </section>
  );
}


export default function App() {
  return (
    <>
      <header className="app-header">
        <h1>
          <span>Query Library</span>
        </h1>
      </header>
      <main className="dashboard">
        <ActivityFeed />
      </main>
    </>
  );
}
