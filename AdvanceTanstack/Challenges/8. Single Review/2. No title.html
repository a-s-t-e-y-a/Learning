Hint #1

Creating a custom hook to fetch an individual review is a good way to go. Here's an example of what that might look like:

function useReview(reviewId = 2) {
  return useQuery({
    queryKey: ["reviews", reviewId],
    queryFn: () => getReviewById(reviewId)
  });
}
Hint #2

Since we are using data from the review query to fetch the book details, we need to make sure the book query is only enabled when the review query has successfully fetched the review details. We can do this by passing the enabled option to the useQuery hook.

function useBookDetails(bookId) {
  return useQuery({
    queryKey: ["books", bookId],
    queryFn: () => getBook(bookId),
    enabled: Boolean(bookId)
  });
}
Hint #3

You can take the data returned from the useReview query and pass it to the useBookDetails custom hook to execute a dependent query. It's probably best to not destructure the result from the useQuery calls because they would have the same name.

const reviewQuery = useReview();
const bookQuery = useBookDetails(reviewQuery.data?.bookId);


if (bookQuery.isError || reviewQuery.isError) {
  return <Error />;
}
Hint #4

When rendering the thumbnail, we need to make sure that data exists before we try to access it. The check needs to be done right before rendering the img component so that the reviews are still rendered as soon as they are available.

{bookQuery.data ? (
  <img src={bookQuery.data.thumbnail} alt={bookQuery.data.title} />
) : null}

import * as React from "react";
import { useQuery } from "@tanstack/react-query";
import { getBook, getReviewById, createStarString } from "./utils";
import { Loading, Error } from "./MessageComponents";


function useReview(reviewId = 2) {
  return useQuery({
    queryKey: ["reviews", reviewId],
    queryFn: () => getReviewById(reviewId)
  });
}


function useBookDetails(bookId) {
  return useQuery({
    queryKey: ["books", bookId],
    queryFn: () => getBook(bookId),
    enabled: Boolean(bookId)
  });
}


function ReviewDetail() {
  const reviewQuery = useReview();
  const bookQuery = useBookDetails(reviewQuery.data?.bookId);


  if (bookQuery.isError || reviewQuery.isError) {
    return <Error />;
  }


  if (reviewQuery.data) {
    return (
      <main className="book-detail">
        <div>
          <span className="book-cover">
            {bookQuery.data ? (
              <img src={bookQuery.data.thumbnail} alt={bookQuery.data.title} />
            ) : null}
          </span>
        </div>
        <div className="reviews">
          <h2>Review</h2>
          <ul>
            <li key={reviewQuery.data.reviewId}>
              <h3>{reviewQuery.data.title}</h3>
              <small>by Anonymous</small>
              <span className="book-rating">
                {createStarString(reviewQuery.data.rating)}
              </span>
              <p>{reviewQuery.data.text}</p>
            </li>
          </ul>
        </div>
      </main>
    );
  }


  return <Loading />;
}


export default function App() {
  return (
    <div>
      <header className="app-header">
        <h1>
          <span>Query Library</span>
        </h1>
      </header>
      <ReviewDetail />
    </div>
  );
}
