Hint #1

In React Query, the QueryClient needs to be stable across renders. The best way to accomplish this is to instantiate the QueryClient outside of the App component.

const queryClient = new QueryClient();


export default function App() {
  // your app code goes here
}
Hint #2

The QueryClientProvider is a React Context Provider that allows you to provide the QueryClient to your app. It should wrap your entire app.

const queryClient = new QueryClient();
export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      {/* the rest of your app code goes here */}
    </QueryClientProvider>
  )
}
Hint #3

The useQuery hook has this signature:

const { data } = useQuery({
  queryKey: ["some-key"],
  queryFn: () => {
    // a function that returns a Promise
  },
});

import {
  QueryClient,
  QueryClientProvider,
  useQuery
} from "@tanstack/react-query";


function Book() {
  const { data } = useQuery({
    queryKey: ["book"],
    queryFn: () => Promise.resolve("The Hobbit")
  });


  return (
    <div>
      <header className="app-header">
        <h1>
          <span>Query Library</span>
        </h1>
      </header>
      <main>
        <h2 className="book-title">{data}</h2>
      </main>
    </div>
  );
}


const queryClient = new QueryClient();


export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <Book />
    </QueryClientProvider>
  );
}

return (

</h1>
</header>

</main>
</div>
);
}

export default function App() {
return <Book />;
}
